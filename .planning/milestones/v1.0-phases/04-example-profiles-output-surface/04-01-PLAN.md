**Goja landmines (encoded from CONTEXT D-19 + RESEARCH §5 Pitfalls 2 & 3):**
- ❌ `new URL(baseUrl)` — goja has no URL global. Use string ops only.
- ❌ `Buffer.from(...)` — goja has no Buffer. Use `String` / `JSON.stringify` only.
- ❌ `process.env` / `process.cwd()` — no `process` in goja. Closures + `__ENV` only.
- ❌ `import fs from 'node:fs'` / `path.join` — no Node modules in goja. String concatenation only.
- ✅ `JSON.stringify(data, null, 2)` — goja-safe.
- ✅ `new Date().toISOString()` — goja-safe.
- ✅ `Object.entries(...)` / optional chaining `?.` / `??` — goja-safe (ES2020 supported).
- ✅ Template literals — goja-safe.
- ✅ Forward-slash paths in handleSummary return keys (`reports/x.md` — NOT `reports\\x.md`).

      profile: string;
      scenario: string;

    function pickActualForBound(metric: MetricEntry | undefined, bound: string): string {
      const values = metric?.values;
      if (!values) return 'n/a';

      return 'n/a';

    }

      lines.push('|---|---|---|---|');

          const actual = pickActualForBound(metric, bound);
          const verdict = result?.ok ? '✅ PASS' : '❌ FAIL';
          lines.push(`| \`${metricName}\` | \`${bound}\` | ${actual} | ${verdict} |`);
        }
      }
      lines.push('');

      // Section 4 — Key metrics
      lines.push('## Key metrics');
      lines.push('');
      lines.push('| Metric | Value |');
      lines.push('|---|---|');
